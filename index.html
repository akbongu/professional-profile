class Neo4jRAGSystem:
    """Neo4j hybrid RAG: graph + vector retrieval for Q&A"""
    def _setup_models(self):
        self.chat = ChatOpenAI(model="gpt-4o-mini", temperature=0)
        self.embeddings = OpenAIEmbeddings()
        self.llm_transformer = LLMGraphTransformer(llm=self.chat)

    def _setup_chain(self):
        # Hybrid: structured graph query + unstructured vector search
        self.chain = (
            RunnableParallel({
                "context": self.retriever | self.format_docs,
                "question": RunnablePassthrough()
            }) | self.prompt | self.chat | StrOutputParser()
        )

// Auto-resolve latest Amazon Linux 2023 AMI via SSM
async function getLatestAMI(region: string): Promise<string> {
  const cmd = new GetParameterCommand({
    Name: "/aws/service/ami-amazon-linux-latest/al2023-ami-kernel-default-x86_64"
  });
  const res = await ssmClient.send(cmd);
  return res.Parameter?.Value ?? fallbackAMIs[region]; // 8 region fallback map
}
